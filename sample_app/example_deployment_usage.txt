# DEPLOY
gcloud functions deploy zcor --entry-point cloud_function_entry_point --runtime python39 --trigger-http --allow-unauthenticated --memory 2048MB --region us-central1 

# REGISTER API KEY
curl http://localhost:5000/generate_api_key

# PROCESS THE JSON
curl -X POST http://localhost:5000/process?fpr_setting=0.5 \
     -H "Content-Type: application/json" \
     -H "x-api-key: <YOUR_API_KEY>" \
     -d '[{"data": "value1"}, {"data": "value2"}]'
     
# CHECK THE AMOUNT DUE
curl http://localhost:5000/usage \
     -H "x-api-key: <YOUR_API_KEY>"
     
# DELETE THE APP
gcloud functions delete zcor
